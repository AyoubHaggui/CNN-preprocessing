from sklearn.model_selection import GridSearchCV
from sklearn.ensemble import RandomForestRegressor
def gridSearch(param_grid):
        forest_reg = RandomForestRegressor()
        grid_search = GridSearchCV(
            forest_reg,
            param_grid,
            cv=2,
            scoring='neg_mean_squared_error'
            )
        grid_search.fit(train, binary_labels)
        return grid_search

def ShowGridSearchResult(grid_search):
    cvres = grid_search.cv_results_
    #Create dataframe from grid search result for pretty priting
    scores_data_frame = DataFrame(data = cvres)
    #Take only mean test score and associated params rows
    scores_data_frame = scores_data_frame[['mean_test_score', 'params']]
    #order rows by means test sore
    scores_data_frame.sort_values(by=['mean_test_score'], inplace = True, ascending = False)
    #Disable cell content truncating to show the full content
    pd.set_option('display.max_colwidth', -1)
